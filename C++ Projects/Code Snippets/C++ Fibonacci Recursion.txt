#include<iostream>
using namespace std;

int factorial(int n);

int main()
{
    int n;

    cout << "Enter a positive integer: ";
    cin >> n;

    cout << "Factorial of " << n << " = " << factorial(n);

    return 0;
}

int fib(int n){  
if (n <= 2) return 1;
   else return fib(n - 1) + fib(n - 2);
}
